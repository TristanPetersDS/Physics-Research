% ---- Algorithm (Paper B), as numbered steps ------------------------
\begin{enumerate}
    \item Generate reference data for every incident angle by rotating the
          grid overlay on simulated events
    \item Bin prompt--delayed displacement vectors into matrices, counting
          the voxels separating the IBD vertex and the capture vertex;
          normalize each reference matrix
    \item Build and normalize the binning matrix of the data set
          (unknown angle)
    \item Compute the \(L^2\) norm of the matrix difference at each
          reference angle \(\vartheta\)~[1]:
    \begin{equation}
      \boxed{\;L^2_i(\vartheta) = \sqrt{\sum_{k,j}\,(x_{kj})_i^2}\;}
      \qquad (x_{kj})_i \in [\mathrm{DifferenceSet}_i](\vartheta)
      \label{eq:l2norm}
    \end{equation}
    \item Fit \(\alpha\left|\sin\left(\frac{\vartheta-\vartheta_0}{2}\right)\right|\)
          to the \(L^2\) norms as a function of angle; the minimum gives the
          source direction \(\vartheta_0\)
\end{enumerate}
